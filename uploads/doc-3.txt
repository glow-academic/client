// CS180 Lab: Polymorphism and Dynamic Binding
// Scenario: A freshman is confused about method overriding and dynamic dispatch

/*
Lab Prompt:
Create a class hierarchy where Animals make sounds. Demonstrate polymorphism using method overriding.
Use base class Animal and subclasses: Dog, Cat, Cow.
Use dynamic dispatch to call makeSound() through Animal references.
*/

class Animal {
    public void makeSound() {
        System.out.println("Some generic animal sound");
    }
}

class Dog extends Animal {
    @Override
    public void makeSound() {
        System.out.println("Woof!");
    }
}

class Cat extends Animal {
    @Override
    public void makeSound() {
        System.out.println("Meow!");
    }
}

class Cow extends Animal {
    @Override
    public void makeSound() {
        System.out.println("Moo!");
    }
}

public class PolymorphismLab {
    public static void main(String[] args) {
        Animal[] zoo = { new Dog(), new Cat(), new Cow() };
        for (Animal a : zoo) {
            a.makeSound(); // Dynamic binding: correct method called at runtime
        }
    }
}
